## Usage

### Installation

<InstallBlock command="animated-gradient" />

### Basic Example

```tsx
import { AnimatedGradient } from "@/components/animated-gradient";

export default function App() {
  return (
    <AnimatedGradient
      variant="mist"
      speed={1}
      opacity={0.8}
      className="h-screen w-full"
    />
  );
}
```

## Custom Configuration

Use the `custom` variant to define organic, warped flows with your own custom shader parameters in real-time.

```tsx
import { AnimatedGradient } from "@/components/animated-gradient";

export default function App() {
  return (
    <AnimatedGradient
      variant="custom"
      config={{
        color1: "#ff007f",
        color2: "#7f00ff",
        color3: "#0000ff",
        direction: 45,
        speed1: 15,
        speed2: 50,
      }}
      className="h-screen w-full"
    />
  );
}
```

## Properties

<PropsTable>
  <Prop
    name="variant"
    type='"mist" | "lava" | "prism" | "plasma" | "pulse" | "vortex" | "custom"'
    default="mist"
    description="Visual effect variant. Each creates a unique animated pattern."
  />
  <Prop
    name="speed"
    type="number"
    default="1"
    description="Animation speed multiplier. Higher values = faster animation."
  />
  <Prop
    name="opacity"
    type="number"
    default="0.8"
    description="Canvas opacity (0-1). Use to blend with background layers."
  />
  <Prop
    name="config"
    type="CustomConfig"
    default="—"
    description="Custom shader configuration. Only applicable when variant is 'custom'. See CustomConfig options below."
  />
  <Prop
    name="className"
    type="string"
    default="—"
    description="Additional CSS classes for the container."
  />
  <Prop
    name="children"
    type="ReactNode"
    default="—"
    description="Overlay content to display on top of the gradient background."
  />
</PropsTable>

### CustomConfig Options

The `config` property accepts an object of type `CustomConfig` with the following parameters:

  <Prop
    name="color1"

  <Prop
    name="color2"

  />
  <Prop
    name="direction"

    default="0"
    description="Flow rotation angle in degrees (0 to 360)."
  />
  <Prop
    name="speed1"

    description="Flow speed multiplier for the first primary warping noise layer."
  />
  <Prop
    name="speed2"

    description="Flow speed multiplier for the second warping noise layer."
  />
  <Prop

  <Prop
    name="frequency2"

  />

    default="10"

  />
  <Prop
    name="amplitude1"
